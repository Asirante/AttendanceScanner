utils/data_collector.py
"""동영상/웹캠 → MTCNN 필터링 → raw float 텐서(.pt) 저장.

저장 규칙:
  - post_process=False 로 raw float(0~255 범위) 텐서를 저장한다.
  - 정규화 (x-127.5)/128.0 은 Dataset / registry 에서 '한 번만' 수행한다.
    (수집 단계와 학습 단계에서 중복 정규화하면 범위가 틀어진다.)
"""
import os

import cv2
import torch
from facenet_pytorch import MTCNN
from PIL import Image


class FaceDataCollector:
    def __init__(self, emp_id: str, save_dir: str, conf_thr: float = 0.95):
        self.emp_id = emp_id
        self.save_dir = save_dir
        self.conf_thr = conf_thr
        device = "cuda" if torch.cuda.is_available() else "cpu"
        # image_size=160: InceptionResnetV1 입력 크기 고정
        # margin=20: 얼굴 경계에 여백 추가 → 정합성 향상
        # post_process=False: [-1,1] 정규화는 나중에 직접 수행(중복 방지)
        self.mtcnn = MTCNN(
            image_size=160,
            margin=20,
            keep_all=False,
            post_process=False,
            device=device,
        )

    def _save_face(self, img_pil: Image.Image, out_dir: str, saved: int) -> bool:
        """검출/필터/저장 공통 로직. 저장 성공 시 True."""
        boxes, probs = self.mtcnn.detect(img_pil)
        if probs is not None and probs[0] is not None and probs[0] >= self.conf_thr:
            face_tensor = self.mtcnn(img_pil)  # (3, 160, 160) raw float
            if face_tensor is not None:
                path = os.path.join(out_dir, f"{saved:04d}.pt")
                torch.save(face_tensor, path)
                return True
        return False

    def collect_from_video(self, video_path: str, frame_skip: int = 5) -> int:
        """동영상 → MTCNN 필터링 → .pt 텐서 저장. 저장 프레임 수 반환."""
        cap = cv2.VideoCapture(video_path)
        out_dir = os.path.join(self.save_dir, self.emp_id)
        os.makedirs(out_dir, exist_ok=True)
        saved, frame_idx = 0, 0
        while cap.isOpened():
            ret, frame = cap.read()
            if not ret:
                break
            if frame_idx % frame_skip != 0:  # Frame Skipping
                frame_idx += 1
                continue
            img_pil = Image.fromarray(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))
            if self._save_face(img_pil, out_dir, saved):
                saved += 1
            frame_idx += 1
        cap.release()
        return saved  # 권장: 직원 1인 ≥ 50프레임

    def collect_live(self, duration_sec: int = 8) -> int:
        """웹캠 라이브 캡처 버전 — 동영상 파일 없이 직접 수집."""
        cap = cv2.VideoCapture(0)
        out_dir = os.path.join(self.save_dir, self.emp_id)
        os.makedirs(out_dir, exist_ok=True)
        saved, frame_idx = 0, 0
        start = cv2.getTickCount()
        while cap.isOpened():
            elapsed = (cv2.getTickCount() - start) / cv2.getTickFrequency()
            if elapsed > duration_sec:
                break
            ret, frame = cap.read()
            if not ret:
                break
            if frame_idx % 5 == 0:
                img_pil = Image.fromarray(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))
                if self._save_face(img_pil, out_dir, saved):
                    saved += 1
            frame_idx += 1
        cap.release()
        return saved


if __name__ == "__main__":
    # 사용 예시: 동영상에서 수집
    #   python -m utils.data_collector
    collector = FaceDataCollector(emp_id="E001", save_dir="data/raw")
    n = collector.collect_from_video("data/videos/E001.mp4", frame_skip=5)
    print(f"E001: {n} 프레임 저장")
utils/split_dataset.py · 70/15/15 분할
"""raw_dir/{emp_id}/*.pt → out_dir/{train|val|test}/{emp_id}/*.pt  (70/15/15 분할).

[FIX #1] 원본의 `if len(frames) 10:` 구문 오류(연산자 누락)를 수정.
         프레임이 최소 개수 미만이면 경고 후 skip 한다.
"""
import glob
import os
import shutil

from sklearn.model_selection import train_test_split

MIN_FRAMES = 10  # 이 미만이면 분할 의미가 없으므로 skip


def split_dataset(
    raw_dir: str,
    out_dir: str,
    train_ratio: float = 0.70,
    val_ratio: float = 0.15,
    min_frames: int = MIN_FRAMES,
):
    for emp_id in sorted(os.listdir(raw_dir)):
        emp_path = os.path.join(raw_dir, emp_id)
        if not os.path.isdir(emp_path):
            continue
        frames = glob.glob(os.path.join(emp_path, "*.pt"))

        # [FIX #1] 연산자 누락 수정: 프레임 부족 시 경고 후 skip
        if len(frames) < min_frames:
            print(f"[경고] {emp_id}: 프레임 {len(frames)}개 — 추가 수집 권장 (skip)")
            continue

        train_f, tmp = train_test_split(
            frames, test_size=1 - train_ratio, random_state=42
        )
        # tmp 를 val/test 로 절반씩 분할 (기본 15/15)
        val_f, test_f = train_test_split(tmp, test_size=0.5, random_state=42)

        for split, files in [("train", train_f), ("val", val_f), ("test", test_f)]:
            dest = os.path.join(out_dir, split, emp_id)
            os.makedirs(dest, exist_ok=True)
            for f in files:
                shutil.copy2(f, dest)
        print(f"  {emp_id}: train={len(train_f)} / val={len(val_f)} / test={len(test_f)}")


if __name__ == "__main__":
    split_dataset(raw_dir="data/raw", out_dir="data/splits")

utils/registry.py · 직원 등록 & 임베딩 저장
"""직원 등록 & 임베딩 DB + label_map 유틸.

인식은 분류기(Classifier)가 아니라 코사인 유사도로 수행한다.
신규 직원은 임베딩만 DB에 INSERT 하면 재학습 없이 즉시 적용된다.

[FIX #7] 등록 임베딩은 항상 FP32 로 계산/저장한다.
         (추론은 FP16 가속을 쓰더라도 등록과 정밀도를 일치시켜 threshold 근처
          오인식을 줄인다. 등록은 1회성이라 FP32 비용이 무의미하다.)
[FIX #6] find_best_match 는 best_sim 초기값을 -1.0(코사인 최솟값)으로 두어
         모든 유사도가 음수인 경우에도 가장 높은 후보를 정확히 선택한다.
"""
import json
import os
import sqlite3

import cv2
import numpy as np
import torch
from facenet_pytorch import MTCNN, InceptionResnetV1
from PIL import Image

from . import DB_PATH


def _get_backbone(device):
    return InceptionResnetV1.from_pretrained("vggface2").eval().to(device)


def _ensure_employees_table(conn: sqlite3.Connection):
    conn.executescript(
        """
        CREATE TABLE IF NOT EXISTS Employees (
            emp_id    TEXT  PRIMARY KEY,
            name      TEXT  NOT NULL,
            embedding BLOB  NOT NULL
        );
        """
    )
    conn.commit()


def capture_embedding(
    emp_id: str,
    name: str,
    duration_sec: int = 5,
    n_frames: int = 10,
):
    """웹캠으로 얼굴 촬영 → 평균 임베딩(FP32) 계산 → Employees 테이블에 저장.

    duration_sec 동안 최대 n_frames 개 임베딩을 평균내어 노이즈를 줄인다.
    """
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    mtcnn = MTCNN(image_size=160, margin=20, post_process=False, device=device)
    backbone = _get_backbone(device)  # [FIX #7] FP32 그대로 사용 (half() 호출 안 함)

    cap = cv2.VideoCapture(0)
    embeddings, collected = [], 0
    start = cv2.getTickCount()
    print(f"[{emp_id}] 촬영 시작 — {duration_sec}초 동안 정면을 바라보세요.")

    # [FIX] 원본에서 끊겨 있던 while 조건 복원
    while cap.isOpened() and collected < n_frames:
        elapsed = (cv2.getTickCount() - start) / cv2.getTickFrequency()
        if elapsed > duration_sec:
            break
        ret, frame = cap.read()
        if not ret:
            break
        img_pil = Image.fromarray(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))
        boxes, probs = mtcnn.detect(img_pil)
        if probs is not None and probs[0] is not None and probs[0] >= 0.95:
            face = mtcnn(img_pil)
            if face is not None:
                face = ((face - 127.5) / 128.0).unsqueeze(0).to(device)
                with torch.no_grad():
                    emb = backbone(face)  # (1, 512) FP32
                    emb = torch.nn.functional.normalize(emb, p=2, dim=1)  # L2 정규화
                embeddings.append(emb.squeeze(0).cpu())
                collected += 1
                print(f"  {collected}/{n_frames} 수집", end="\r")
        cv2.imshow("직원 등록 — q로 중단", frame)
        if cv2.waitKey(1) & 0xFF == ord("q"):
            break

    cap.release()
    cv2.destroyAllWindows()

    if not embeddings:
        raise RuntimeError("임베딩 수집 실패 — 얼굴이 감지되지 않았습니다.")

    # n_frames 개 평균 → 대표 임베딩, 다시 L2 정규화
    mean_emb = torch.stack(embeddings).mean(dim=0)
    mean_emb = torch.nn.functional.normalize(mean_emb, p=2, dim=0)
    emb_bytes = mean_emb.numpy().astype(np.float32).tobytes()  # FP32 BLOB 직렬화

    conn = sqlite3.connect(DB_PATH)
    _ensure_employees_table(conn)
    conn.execute(
        "INSERT OR REPLACE INTO Employees (emp_id, name, embedding) VALUES (?, ?, ?)",
        (emp_id, name, emb_bytes),
    )
    conn.commit()
    conn.close()
    print(f"\n[완료] {name}({emp_id}) 등록됨 — {collected}프레임 평균 임베딩 저장")


def load_all_embeddings() -> dict:
    """DB에서 전체 직원 임베딩 로드.

    반환: {emp_id: {"name": str, "embedding": torch.Tensor(512,) FP32}}
    main.py 시작 시 1회 호출하여 메모리에 캐싱한다.
    """
    conn = sqlite3.connect(DB_PATH)
    rows = conn.execute("SELECT emp_id, name, embedding FROM Employees").fetchall()
    conn.close()
    result = {}
    for emp_id, name, emb_blob in rows:
        emb = torch.from_numpy(np.frombuffer(emb_blob, dtype=np.float32).copy())
        result[emp_id] = {"name": name, "embedding": emb}
    return result


def find_best_match(
    query_emb: torch.Tensor,
    db_embeddings: dict,
    threshold: float = 0.70,
) -> tuple:
    """query_emb: 실시간 추론에서 얻은 (512,) L2-정규화 FP32 벡터.

    반환: (emp_id, name, similarity) 또는 (None, "unknown", best_sim)
    """
    best_id, best_sim = None, -1.0  # [FIX #6] 코사인 최솟값으로 초기화
    for emp_id, data in db_embeddings.items():
        # 양쪽 모두 L2 정규화되어 있으므로 내적 = 코사인 유사도
        sim = torch.dot(query_emb, data["embedding"]).item()
        if sim > best_sim:
            best_sim, best_id = sim, emp_id
    if best_id is not None and best_sim >= threshold:
        return best_id, db_embeddings[best_id]["name"], best_sim
    return None, "unknown", best_sim


# ── label_map 유틸 (분류기 학습/평가에서 사용) ──────────────────────────
def build_label_map(splits_train_dir: str = "data/splits/train",
                    out_path: str = "data/label_map.json") -> dict:
    """train 디렉토리의 emp_id 폴더를 정렬하여 {idx: emp_id} 매핑 생성/저장.

    핵심: 인덱스 ↔ emp_id 매핑은 학습/평가 전 구간에서 일관되게 유지되어야 한다.
    """
    emp_ids = sorted(
        d for d in os.listdir(splits_train_dir)
        if os.path.isdir(os.path.join(splits_train_dir, d))
    )
    label_map = {idx: emp_id for idx, emp_id in enumerate(emp_ids)}
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(label_map, f, ensure_ascii=False, indent=2)
    print(f"label_map 저장: {out_path} ({len(label_map)}명)")
    return label_map


def load_label_map(path: str = "data/label_map.json") -> dict:
    """{int_idx: emp_id} 형태로 로드 (JSON 키는 문자열이므로 int 캐스팅)."""
    with open(path, "r", encoding="utf-8") as f:
        raw = json.load(f)
    return {int(k): v for k, v in raw.items()}

"""학습용 분류기 모델.

주의: 실제 운영 인식은 main.py 의 코사인 유사도 방식이 담당한다.
이 분류기는 backbone 품질을 검증/실험(EXP-1~6)하기 위한 용도다.
"""
import torch
import torch.nn as nn
from facenet_pytorch import InceptionResnetV1


class FaceClassifier(nn.Module):
    def __init__(self, num_classes: int = 30, freeze_backbone: bool = True):
        super().__init__()
        # pretrained='vggface2': VGGFace2 사전학습 가중치 (512-dim 임베딩)
        self.backbone = InceptionResnetV1.from_pretrained("vggface2")
        self.classifier = nn.Linear(512, num_classes)
        if freeze_backbone:
            for p in self.backbone.parameters():
                p.requires_grad = False  # Feature Extractor 전면 동결

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        embedding = self.backbone(x)  # (B, 512)  ← frozen
        return self.classifier(embedding)  # (B, num_classes) ← trainable

    def unfreeze_last_blocks(self, n: int = 2):
        """EXP-2: 마지막 n개 block 만 Fine-tuning 해동."""
        blocks = list(self.backbone.named_children())
        for name, module in blocks[-n:]:
            for p in module.parameters():
                p.requires_grad = True
            print(f"Unfrozen block: {name}")

    def get_embedding(self, x: torch.Tensor) -> torch.Tensor:
        """분류 없이 512-dim 임베딩만 추출."""
        with torch.no_grad():
            return self.backbone(x)

"""PyTorch Dataset — raw float .pt 텐서를 로드하여 학습용 정규화 텐서 반환.

[FIX #2] 증강(ColorJitter/Flip)을 '정규화 이전'에 적용한다.
         - 저장된 텐서는 raw float(0~255 범위)이다.
         - ColorJitter 는 [-1,1] 음수 범위에서 비정상 동작하므로,
           먼저 [0,1] 로 스케일 → 증강 → 그 다음 InceptionResnet 범위([-1,1])로 정규화.
"""
import os

import torch
from torch.utils.data import Dataset
from torchvision import transforms


class FaceDataset(Dataset):
    def __init__(self, root_dir: str, label_map: dict, augment: bool = False):
        """root_dir: data/splits/{train|val|test}/
        label_map: {int_idx: emp_id}  ← registry.load_label_map()
        """
        self.samples = []
        # idx ↔ emp_id 역방향 매핑 (emp_id → idx)
        reverse_map = {v: k for k, v in label_map.items()}

        for emp_id in sorted(os.listdir(root_dir)):
            if emp_id not in reverse_map:
                continue
            idx = reverse_map[emp_id]
            emp_dir = os.path.join(root_dir, emp_id)
            if not os.path.isdir(emp_dir):
                continue
            for fname in os.listdir(emp_dir):
                if fname.endswith(".pt"):
                    self.samples.append((os.path.join(emp_dir, fname), idx))

        # 증강은 학습 split 에만 적용 (val/test 는 augment=False)
        # ColorJitter 는 [0,1] 범위 입력을 기대하므로 정규화 '이전'에 적용한다.
        self.aug = (
            transforms.Compose(
                [
                    transforms.RandomHorizontalFlip(p=0.5),
                    transforms.ColorJitter(
                        brightness=0.2, contrast=0.2, saturation=0.1
                    ),
                ]
            )
            if augment
            else None
        )

    def __getitem__(self, idx):
        path, label = self.samples[idx]
        tensor = torch.load(path)  # (3, 160, 160) raw float (0~255 범위)

        if self.aug is not None:
            # [FIX #2] 정규화 이전, [0,1] 범위에서 증강
            tensor = (tensor / 255.0).clamp(0.0, 1.0)
            tensor = self.aug(tensor)
            # 증강 후 InceptionResnet 입력 범위 [-1,1] 로 변환
            tensor = tensor * 2.0 - 1.0
        else:
            # 증강 없을 때는 기존과 동일한 정규화: (x-127.5)/128 ≈ [-1,1]
            tensor = (tensor - 127.5) / 128.0

        return tensor, label

    def __len__(self):
        return len(self.samples)

"""분류기 학습 루프 (EXP-1~6 실험용).

label_map.json 이 없으면 자동 생성한다.
"""
import os

import torch
import torch.nn as nn
from torch.optim.lr_scheduler import CosineAnnealingLR
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

from models.classifier import FaceClassifier
from utils.dataset import FaceDataset
from utils.registry import build_label_map, load_label_map


def evaluate(model, loader, device) -> float:
    model.eval()
    correct, total = 0, 0
    with torch.no_grad():
        for imgs, labels in loader:
            imgs, labels = imgs.to(device), labels.to(device)
            preds = model(imgs).argmax(1)
            correct += (preds == labels).sum().item()
            total += labels.size(0)
    return correct / total if total else 0.0


def train(config: dict):
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    writer = SummaryWriter(f"runs/{config['exp_name']}")

    label_map_path = "data/label_map.json"
    if not os.path.exists(label_map_path):
        build_label_map(config["train_dir"], label_map_path)
    lmap = load_label_map(label_map_path)

    train_ds = FaceDataset(config["train_dir"], lmap, augment=True)
    val_ds = FaceDataset(config["val_dir"], lmap, augment=False)
    tl = DataLoader(train_ds, batch_size=32, shuffle=True, num_workers=4, pin_memory=True)
    vl = DataLoader(val_ds, batch_size=32, shuffle=False, num_workers=4, pin_memory=True)

    model = FaceClassifier(
        num_classes=len(lmap), freeze_backbone=config["freeze"]
    ).to(device)
    if config.get("unfreeze_blocks"):
        model.unfreeze_last_blocks(config["unfreeze_blocks"])

    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(
        filter(lambda p: p.requires_grad, model.parameters()),
        lr=config["lr"],
        weight_decay=1e-4,
    )
    scheduler = CosineAnnealingLR(optimizer, T_max=config["epochs"], eta_min=1e-6)

    best_val, patience, no_imp = 0.0, 10, 0
    for epoch in range(config["epochs"]):
        model.train()
        t_loss = 0.0
        for imgs, labels in tl:
            imgs, labels = imgs.to(device), labels.to(device)
            optimizer.zero_grad()
            loss = criterion(model(imgs), labels)
            loss.backward()
            optimizer.step()
            t_loss += loss.item()

        val_acc = evaluate(model, vl, device)
        scheduler.step()

        writer.add_scalar("Loss/train", t_loss / len(tl), epoch)
        writer.add_scalar("Acc/val", val_acc, epoch)
        writer.add_scalar("LR", optimizer.param_groups[0]["lr"], epoch)
        print(
            f"Epoch {epoch + 1:3d} | loss={t_loss / len(tl):.4f} | val_acc={val_acc:.4f}"
        )

        if val_acc > best_val:
            best_val, no_imp = val_acc, 0
            torch.save(model.state_dict(), f"models/{config['exp_name']}_best.pth")
        else:
            no_imp += 1
            if no_imp >= patience:
                print(f"Early stopping @ epoch {epoch + 1}")
                break

    writer.close()
    print(f"Best val_acc: {best_val:.4f}")


if __name__ == "__main__":
    train(
        {
            "exp_name": "exp01_transfer_frozen",
            "train_dir": "data/splits/train",
            "val_dir": "data/splits/val",
            "epochs": 50,
            "lr": 1e-3,
            "freeze": True,
            "unfreeze_blocks": None,
        }
    )

"""Test set 최종 평가 + Inference Time 측정.

classification_report / confusion_matrix / FPS 를 출력한다.
"""
import time

import torch
from sklearn.metrics import classification_report, confusion_matrix
from torch.utils.data import DataLoader

from models.classifier import FaceClassifier
from utils.dataset import FaceDataset
from utils.registry import load_label_map


def run_evaluation(model_path: str, test_dir: str):
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    lmap = load_label_map()  # {idx: emp_id}
    test_ds = FaceDataset(test_dir, lmap, augment=False)
    # batch=1: 정확한 per-frame latency 측정
    loader = DataLoader(test_ds, batch_size=1, shuffle=False)

    model = FaceClassifier(num_classes=len(lmap)).to(device)
    model.load_state_dict(torch.load(model_path, map_location=device))
    model.eval()

    all_preds, all_labels, latencies = [], [], []
    with torch.no_grad():
        for imgs, labels in loader:
            imgs = imgs.to(device)
            if device.type == "cuda":
                torch.cuda.synchronize()
            t0 = time.perf_counter()
            preds = model(imgs).argmax(1)
            if device.type == "cuda":
                torch.cuda.synchronize()
            latencies.append((time.perf_counter() - t0) * 1000)  # ms
            all_preds.extend(preds.cpu().tolist())
            all_labels.extend(labels.tolist())

    # ── 성능 리포트 ─────────────
    target_names = [lmap[i] for i in range(len(lmap))]
    print(classification_report(all_labels, all_preds, target_names=target_names))
    print("Confusion Matrix:\n", confusion_matrix(all_labels, all_preds))

    avg_ms = sum(latencies) / len(latencies)
    fps = 1000 / avg_ms
    print(f"\nInference Time: {avg_ms:.2f} ms/frame  ({fps:.1f} FPS)")
    print(f"목표 FPS ≥15: {'✓ 달성' if fps >= 15 else '✗ 미달 — 최적화 필요'}")


if __name__ == "__main__":
    run_evaluation("models/exp01_transfer_frozen_best.pth", "data/splits/test")

"""SQLite 근태 DB + 상태 머신.

상태 전이:
  None(오늘 첫 방문) → 출근
  출근              → 외출 | 퇴근
  외출              → 복귀
  복귀              → 외출 | 퇴근
  퇴근              → (없음, 당일 종료)

마지막 상태는 ORDER BY log_id DESC 로 조회한다(시간 문자열 정렬에 의존하지 않음).
주의: 자정을 넘기는 야간 근무는 date.today() 기준이라 지원 범위 밖이다.
"""
import os
import sqlite3
from datetime import date, datetime
from typing import List, Optional

from . import DB_PATH


class AttendanceDB:
    def __init__(self, db_path: Optional[str] = None):
        self.db_path = db_path or DB_PATH
        # NAS 경로의 상위 디렉토리가 없으면 즉시 알 수 있게 함
        self.conn = sqlite3.connect(self.db_path, check_same_thread=False)
        self._init_schema()

    def _init_schema(self):
        self.conn.executescript(
            """
            CREATE TABLE IF NOT EXISTS Employees (
                emp_id    TEXT  PRIMARY KEY,
                name      TEXT  NOT NULL,
                embedding BLOB  NOT NULL    -- 512-dim float32 직렬화
            );
            CREATE TABLE IF NOT EXISTS AttendanceLog (
                log_id  INTEGER  PRIMARY KEY AUTOINCREMENT,
                emp_id  TEXT     NOT NULL,
                date    TEXT     NOT NULL,
                time    TEXT     NOT NULL,
                state   TEXT     NOT NULL,
                FOREIGN KEY (emp_id) REFERENCES Employees(emp_id)
            );
            CREATE INDEX IF NOT EXISTS idx_log_emp_date
                ON AttendanceLog(emp_id, date);
            """
        )
        self.conn.commit()

    # 상태 머신 정의 — 단일 출처(single source of truth)
    TRANSITIONS = {
        None: ["출근"],
        "출근": ["외출", "퇴근"],
        "외출": ["복귀"],
        "복귀": ["외출", "퇴근"],
        "퇴근": [],
    }

    def get_last_state(self, emp_id: str) -> Optional[str]:
        today = date.today().isoformat()
        cur = self.conn.execute(
            "SELECT state FROM AttendanceLog "
            "WHERE emp_id=? AND date=? ORDER BY log_id DESC LIMIT 1",
            (emp_id, today),
        )
        row = cur.fetchone()
        return row[0] if row else None

    def get_available_actions(self, emp_id: str) -> List[str]:
        """현재 상태 → 가능한 액션 리스트 반환."""
        return self.TRANSITIONS.get(self.get_last_state(emp_id), [])

    def log_action(self, emp_id: str, state: str):
        """유효성 검사 후 로그 INSERT."""
        valid = self.get_available_actions(emp_id)
        if state not in valid:
            raise ValueError(
                f"유효하지 않은 전이: {self.get_last_state(emp_id)} → {state}"
            )
        now = datetime.now()
        self.conn.execute(
            "INSERT INTO AttendanceLog (emp_id, date, time, state) VALUES (?,?,?,?)",
            (emp_id, now.strftime("%Y-%m-%d"), now.strftime("%H:%M:%S"), state),
        )
        self.conn.commit()

    def get_employee_name(self, emp_id: str) -> str:
        cur = self.conn.execute(
            "SELECT name FROM Employees WHERE emp_id=?", (emp_id,)
        )
        row = cur.fetchone()
        return row[0] if row else emp_id  # 미등록 시 emp_id 반환

    def close(self):
        self.conn.close()

        print(f"[경고] {nas_dir} 가 마운트 지점이 아닙니다 — 로컬 디렉토리로 진행")


def main():
    check_nas_mount()
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    use_fp16 = USE_FP16 and device.type == "cuda"

    mtcnn = MTCNN(image_size=160, margin=20, post_process=False, device=device)
    backbone = InceptionResnetV1.from_pretrained("vggface2").eval().to(device)
    if use_fp16:
        backbone.half()  # FP16: 속도 ~1.5x (비교 전 .float() 로 되돌림)

    db_embeddings = load_all_embeddings()  # {emp_id: {name, embedding(FP32)}}
    print(f"등록 직원 {len(db_embeddings)}명 로드 완료")
    if not db_embeddings:
        print("[경고] 등록된 직원이 없습니다. registry.capture_embedding() 으로 먼저 등록하세요.")

    db = AttendanceDB()
    cap = cv2.VideoCapture(0)

    vote_buf = deque(maxlen=VOTE_N)  # [FIX #4] 최근 N개만 유지
    fps_t = cv2.getTickCount() / cv2.getTickFrequency()

    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break
        draw_silhouette(frame, SILHOUETTE)

        img_pil = Image.fromarray(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))
        boxes, probs = mtcnn.detect(img_pil)

        if boxes is not None and len(boxes) > 0:
            cx = (boxes[0][0] + boxes[0][2]) / 2
            cy = (boxes[0][1] + boxes[0][3]) / 2
            # [FIX] 원본에서 끊겨 있던 in_zone 조건 복원
            in_zone = (
                SILHOUETTE[0] <= cx <= SILHOUETTE[2]
                and SILHOUETTE[1] <= cy <= SILHOUETTE[3]
            )
            if in_zone:
                face_t = mtcnn(img_pil)
                if face_t is not None:
                    inp = ((face_t - 127.5) / 128.0).unsqueeze(0).to(device)
                    if use_fp16:
                        inp = inp.half()
                    with torch.no_grad():
                        emb = backbone(inp)  # (1, 512)
                        emb = torch.nn.functional.normalize(emb, p=2, dim=1)
                        # [FIX #7] 비교는 FP32 로 (등록 임베딩과 정밀도 일치)
                        emb = emb.squeeze(0).float().cpu()

                    emp_id, emp_name, sim = find_best_match(
                        emb, db_embeddings, threshold=SIM_THR
                    )

                    if emp_id is not None:
                        # [FIX #4] 직전과 다른 사람이면 버퍼 초기화 후 새로 시작
                        if vote_buf and vote_buf[-1] != emp_id:
                            vote_buf.clear()
                        vote_buf.append(emp_id)

                        # 최근 VOTE_N개가 모두 동일인일 때만 확정
                        if (
                            len(vote_buf) == VOTE_N
                            and len(set(vote_buf)) == 1
                        ):
                            final_id = vote_buf[0]
                            final_name = db_embeddings[final_id]["name"]
                            actions = db.get_available_actions(final_id)
                            if actions:
                                show_action_popup(
                                    frame, final_name, actions, db, final_id
                                )
                            else:
                                print(f"{final_name}: 당일 퇴근 완료 — 추가 액션 없음")
                            vote_buf.clear()
                    else:
                        vote_buf.clear()  # 임계값 미달 시 버퍼 초기화

        now = cv2.getTickCount() / cv2.getTickFrequency()
        fps = 1.0 / (now - fps_t + 1e-9)
        fps_t = now
        draw_hud(frame, fps)

        cv2.imshow("출퇴근 시스템 (q: 종료)", frame)
        if cv2.waitKey(1) & 0xFF == ord("q"):
            break

    cap.release()
    db.close()
    cv2.destroyAllWindows()


if __name__ == "__main__":
    main()

"""OpenCV UI 헬퍼 — 실루엣 가이드, HUD, 인식 완료 팝업.

[FIX #5] show_action_popup 의 무한 대기(cv2.waitKey(0))를 타임아웃으로 교체.
         사용자가 키를 누르지 않아도 timeout_ms 후 자동 취소되어
         메인 영상 루프가 얼어붙지 않는다.
"""
import cv2
import numpy as np

from .database import AttendanceDB

FONT = cv2.FONT_HERSHEY_SIMPLEX


def draw_silhouette(frame: np.ndarray, rect: tuple):
    """화면 중앙 실루엣 사각형 + 가이드 텍스트 렌더링."""
    x1, y1, x2, y2 = rect
    overlay = frame.copy()
    cv2.rectangle(overlay, (0, 0), (frame.shape[1], frame.shape[0]), (0, 0, 0), -1)
    cv2.rectangle(overlay, (x1, y1), (x2, y2), (255, 255, 255), -1)  # 실루엣 마스크
    cv2.addWeighted(overlay, 0.15, frame, 0.85, 0, frame)  # 반투명 합성
    cv2.rectangle(frame, (x1, y1), (x2, y2), (100, 210, 255), 2)
    cv2.putText(
        frame, "Align your face", (x1, y1 - 10), FONT, 0.55, (100, 210, 255), 1
    )


def draw_hud(frame: np.ndarray, fps: float):
    """FPS + 현재 시각 HUD 오버레이."""
    from datetime import datetime

    ts = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    cv2.putText(frame, f"FPS: {fps:.1f}", (10, 30), FONT, 0.65, (0, 255, 0), 2)
    cv2.putText(frame, ts, (10, 55), FONT, 0.55, (200, 200, 200), 1)


def show_action_popup(
    frame: np.ndarray,
    emp_name: str,
    actions: list,
    db: AttendanceDB,
    emp_id: str,
    timeout_ms: int = 5000,
):
    """인식 완료 팝업: 이름 표시 + 가능한 액션 키 매핑.

    키 바인딩: 1=출근 / 2=외출 / 3=복귀 / 4=퇴근 / ESC=취소
    [FIX #5] timeout_ms 동안 입력이 없으면 자동 취소(루프 정지 방지).
    """
    key_map = {ord("1"): "출근", ord("2"): "외출", ord("3"): "복귀", ord("4"): "퇴근"}
    label_to_key = {v: k for k, v in key_map.items()}

    popup = frame.copy()
    cv2.rectangle(popup, (120, 150), (520, 340), (30, 30, 30), -1)
    cv2.rectangle(popup, (120, 150), (520, 340), (100, 210, 255), 2)
    cv2.putText(
        popup, f"Recognized: {emp_name}", (145, 190), FONT, 0.75, (255, 255, 255), 2
    )
    for i, act in enumerate(actions):
        key_char = chr(label_to_key.get(act, 63))  # 63 = '?'
        cv2.putText(
            popup,
            f"[{key_char}] {act}",
            (145, 230 + i * 36),
            FONT,
            0.65,
            (100, 220, 140),
            2,
        )
    cv2.putText(popup, "[ESC] Cancel", (145, 310), FONT, 0.55, (180, 180, 180), 1)
    cv2.imshow("출퇴근 시스템 (q: 종료)", popup)

    # [FIX #5] 무한 대기 대신 타임아웃. waitKey 는 ms 단위, 0 이하면 무한이므로 보정.
    key = cv2.waitKey(max(1, timeout_ms)) & 0xFF
    if key in key_map and key_map[key] in actions:
        db.log_action(emp_id, key_map[key])
        print(f"{emp_name}: {key_map[key]} 완료")
    # ESC(27) 또는 타임아웃/무효 키 → 아무 동작 없이 복귀
⚠
FP16 사용 시 주의 [수정 #7]: backbone.half() 적용 후 입력 텐서도 .half()로 변환해야 합니다. 단, 등록 임베딩은 FP32로 저장되므로 추론 임베딩도 비교 직전 .float()로 되돌려 정밀도를 일치시킵니다(threshold 근처 오인식 방지). CPU에서는 FP16 미지원이므로 USE_FP16 and device.type == 'cuda' 조건부로 처리합니다. 정밀도 불일치가 우려되면 USE_FP16=False로 두면 등록과 완전히 동일해집니다. FPS가 여전히 부족하면 MTCNN 검출 주기를 3프레임마다 1회로 줄이는 것도 고려하세요.
